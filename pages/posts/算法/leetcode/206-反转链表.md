## 题解
### 1.双指针
```JavaScript
var reverseList = function(head) {
    if(!head || !head.next) return head;
    let temp = null, pre = null, cur = head;
    while(cur){
        let temp = cur.next;
        cur.next = pre;
        pre = cur;
        cur = temp;
    }
    return pre
};
```
### 2.递归
```JavaScript
var reverseList = function(head) {
    return reverse(null, head);
};

var reverse  = function (pre , head){
    if(!head) return pre
    const temp = head.next;
    head.next = pre
    pre = head;
    return  reverse(pre,temp)
}
```
